Disjoint Set Data Structure


Code Implementation 

#define ll long long 
ll Parent[10000],N;
ll rank[10000];

void make_set()
{
	for(ll i=1;i<=N;i++)
	{
		Parent[x]=x;
		rank[x]=0;
	}
}

ll find(ll x)
{	
	if(x!=Parent[x]) 
	{
		Parent[x]=find(Parent[x]);
	}
    return Parent[x];
}

void uunion(ll &x,ll &y){
    ll PX=find(x),PY=find(y);
    if(PX==PY)
    { 
    	return;
    }
    if(rank[PX]>rank[PY])
    {
        Parent[PY]=PX;
    }
    else
    {
        Parent[PX]=PY;
        if(rank[PX]==rank[PY])
        {
            rank[PY]++;
        }
    }
}
